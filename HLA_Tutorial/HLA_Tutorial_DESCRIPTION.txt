   0.a) tar zxvf CERTI_TUTORIAL-x.y.z-Source.tar.gz
   0.b) mkdir build_CERTI_TUTORIAL; cd build_CERTI_TUTORIAL 
   0.c) cmake ../CERTI_TUTORIAL-x.y.z
   0.d) make
